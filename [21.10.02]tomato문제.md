백준 알고리즘 문제를 하루종일 풀었다 아직은 해결이 되지 않았지만 내일도 다시 시도해볼 예정이다
첫번째 코드는 다음과 같다

```js
const fs = require("fs");
const filePath = process.platform === 'linux' ? '/dev/stdin' : "./6_7576.txt";

const input = fs.readFileSync(filePath).toString().trim().split("\n");

const n = parseInt(input[0].split(" ")[0]);
const m = parseInt(input.shift().split(" ")[1]);

const visit = [];
const notVisit = [];
let tomato = 0;

notVisit.push("start");
for(let i = 0; i < m; i++){
    input[i] = input[i].split(" ").map(item => +item);
    for (let j = 0; j < n; j++){
        if(input[i][j] === 1){
            notVisit.push(`${i}-${j}`);
            tomato++;
        }else if(input[i][j] === 0){
            tomato++;
        }
    }
}
notVisit.push("finish");

let result = 0;
let temp
while (notVisit.length !== 0) {
    temp = notVisit.shift();
    if (temp === 'start') {
        notVisit.push('start');
        continue;

    }else if(temp === 'finish') {
        if (notVisit.length === 1){
            break;
        }
        notVisit.push('finish');
        result++;
        continue;
    }

    node = temp.split("-").map(item => +item);
    x = node[0];
    y = node[1];

    if (!visit.includes(`${x}-${y}`)){
        visit.push(`${x}-${y}`);
        input[x][y] = 1;
    }
    // 아래로
    if (x < m-1 && input[x+1][y] === 0){
        notVisit.push(`${x+1}-${y}`);
    }
    // 위로
    if (x > 0 && input[x-1][y] === 0){
        notVisit.push(`${x-1}-${y}`);
    }
    // 오른쪽
    if (y < n-1 && input[x][y+1] === 0){
        notVisit.push(`${x}-${y+1}`);
    }
    // 왼쪽
    if (y > 0 && input[x][y-1] === 0){
        notVisit.push(`${x}-${y-1}`);
    }
    
}
if(tomato === visit.length){
    console.log(result);
}else {
    console.log(-1);
}
```

하지만 여전히 **시간초과였다**

펼침연산자도 배열을 모두 돌아야해서 시간초과가 일어나나 싶어서 이번에는 `array.from()`을 이용해 배열대신 `Set`을 사용해보았다

```js
const fs = require("fs");
const filePath = process.platform === 'linux' ? '/dev/stdin' : "./6_7576.txt";

const input = fs.readFileSync(filePath).toString().trim().split("\n");

const n = parseInt(input[0].split(" ")[0]);
const m = parseInt(input.shift().split(" ")[1]);

const visit = new Set();
let notVisit = new Set();
let tomatoNum = 0;
let result = 0;

for(let i = 0; i < m; i++){
    input[i] = input[i].split(" ").map(item => +item);
    for (let j = 0; j < n; j++){
        if(input[i][j] === 1){
            notVisit.add(`${i}-${j}`);
            tomatoNum++;
        }else if(input[i][j] === 0){
            tomatoNum++;
        }
    }
}

let temp = Array.from(notVisit);
while(true) {
    notVisit.clear();
    temp.forEach(tomato => {
        x = parseInt(tomato.split("-")[0]);
        y = parseInt(tomato.split("-")[1]);

        if (!visit.has(`${x}-${y}`)){
            visit.add(`${x}-${y}`);
            input[x][y] = 1;
        }
        // 아래로
        if (x < m-1 && input[x+1][y] === 0){
            notVisit.add(`${x+1}-${y}`);
        }
        // 위로
        if (x > 0 && input[x-1][y] === 0){
            notVisit.add(`${x-1}-${y}`);
        }
        // 오른쪽
        if (y < n-1 && input[x][y+1] === 0){
            notVisit.add(`${x}-${y+1}`);
        }
        // 왼쪽
        if (y > 0 && input[x][y-1] === 0){
            notVisit.add(`${x}-${y-1}`);
        }
    });
    
    if (notVisit.size === 0){
        break;
    }
    result++;
    temp = Array.from(notVisit);
}
if(tomatoNum === visit.size){
    console.log(result);
}else {
    console.log(-1);
}
```
